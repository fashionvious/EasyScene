const App = () => {
  // 状态管理
  const [selectedModel, setSelectedModel] = useState(MODELS[2]); 
  const [styleMode, setStyleMode] = useState('3d'); 
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState(null);

  // 双语编辑状态
  const [promptEn, setPromptEn] = useState('');
  const [promptZh, setPromptZh] = useState('');
  const [isSyncing, setIsSyncing] = useState(null); // 'en2zh' or 'zh2en'

  // UI 抽屉状态
  const [isModelDrawerOpen, setIsModelDrawerOpen] = useState(false);
  const [isStyleDrawerOpen, setIsStyleDrawerOpen] = useState(false);

  const [images, setImages] = useState({
    clothing: null,
    appearance: null,
    layout: null,
    styleRef: null
  });

  const [userText, setUserText] = useState('');

  const handleImageUpload = (e, type) => {
    const file = e.target.files[0];
    if (file) {
      const reader = new FileReader();
      reader.onloadend = () => {
        setImages(prev => ({ ...prev, [type]: reader.result }));
      };
      reader.readAsDataURL(file);
    }
  };

  const removeImage = (type) => {
    setImages(prev => ({ ...prev, [type]: null }));
  };

  const callGemini = async (payload, retries = 5, delay = 1000) => {
    try {
      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });
      if (!response.ok) throw new Error('API 请求失败');
      return await response.json();
    } catch (err) {
      if (retries > 0) {
        await new Promise(resolve => setTimeout(resolve, delay));
        return callGemini(payload, retries - 1, delay * 2);
      }
      throw err;
    }
  };

  // 生成主逻辑 (使用 JSON Schema 强制返回双语)
  const generatePrompt = async () => {
    setIsLoading(true);
    setError(null);
    try {
      let exactStyleRule = styleMode === 'auto' 
        ? `【风格检测与执行】分析参考图是 3D 写实还是 2D 动漫。若判定为 3D 执行: ${selectedModel.styleRules['3d']}。若为 2D 执行: ${selectedModel.styleRules['2d']}`
        : `【强制风格锁定】必须强制转化为 ${styleMode.toUpperCase()} 风格！严格执行防污染规则: ${selectedModel.styleRules[styleMode]}`;

      const layoutTemplateInstruction = `
      【排版与绝对防出画规范 (CRITICAL)】：
      1. 比例 16:9。严格从左到右依次为：[全身侧面]、[全身正面]、[全身背面]、[最右侧肖像特写]。统一灰色背景。
      2. 强制镜头距离：因为画面中有4个元素，极其容易溢出画布！你必须在提示词中强力加入 "wide angle shot" (广角镜头)、"full body turnaround from head to toe" (从头到脚全身完全可见)、"generous negative space above head and below feet" (头顶和脚底留有大量空白边缘)。
      3. 防裁切强调：极力避免头部或脚部被裁切（no zoomed in, no cropped head）。
      `;

      let systemPrompt = `你是一个熟悉各类 AI 绘画模型官方底层的顶级提示词工程师。
      任务：生成适配 ${selectedModel.name} 的极致提示词。
      
      ${selectedModel.lengthRule}
      
      【防污染边界】：${exactStyleRule}
      【排版与防裁切】：${layoutTemplateInstruction}
      【角色一致性】：固化脸型、发色、瞳色、服装结构。强调 "identical character in all 4 panels"。
      
      返回格式要求：你必须严格返回 JSON 格式数据。不要有任何Markdown包裹。
      {
        "english": "这里填写最终生成的英文提示词（含负面参数等）",
        "chinese": "这里填写对英文提示词的精准中文翻译，方便用户理解修改"
      }`;

      let parts = [];
      parts.push({ text: `你是一个极度严谨的视觉特征拆解专家，请严格遵守【特征隔离原则】：
        1. 看到【服装参考图】时：只提取衣物款式/材质。绝对忽略长相！
        2. 看到【角色面孔图】时：只提取面孔/发型。绝对忽略衣服！
        3. 看到【排版模板图】时：只提取构图。绝对忽略人物长相！
        4. 看到【画风参考图】时：只提取艺术风格、氛围光影。绝对忽略具体内容！
        结合以上特征重组，并确保角色完全在画面内（不被裁切）。返回指定JSON。` });
      
      if (images.clothing) parts.push({ text: "\n=== 【服装参考图】 ===" }, { inlineData: { mimeType: "image/png", data: images.clothing.split(',')[1] } });
      if (images.appearance) parts.push({ text: "\n=== 【角色面孔图】 ===" }, { inlineData: { mimeType: "image/png", data: images.appearance.split(',')[1] } });
      if (images.layout) parts.push({ text: "\n=== 【排版模板图】 ===" }, { inlineData: { mimeType: "image/png", data: images.layout.split(',')[1] } });
      if (images.styleRef) parts.push({ text: "\n=== 【画风参考图】 ===" }, { inlineData: { mimeType: "image/png", data: images.styleRef.split(',')[1] } });
      if (userText.trim()) parts.push({ text: `\n=== 【用户附加文本需求】(必须进行智能润色与视觉特征扩充) ===\n内容：${userText}\n【极其重要的指令】：不要生硬地直接拼凑这些词！你必须进行专业的 AI 绘画视觉联想与特征发散。例如用户输入"20岁"，你需要扩充为"年轻的面庞、青春活力、细腻平滑的肌肤、清澈的眼神"等具体视觉细节；输入"战斗"，需扩充为"紧绷的肌肉、战损的衣服、坚毅的神态、动感的光影"等。请将扩充后的丰富视觉细节完美、自然地融合到最终提示词中。` });
      
      if (parts.length <= 1) throw new Error('请至少上传一张参考图或输入附加提示词需求');

      const payload = {
        contents: [{ parts }],
        systemInstruction: { parts: [{ text: systemPrompt }] },
        generationConfig: {
          responseMimeType: "application/json",
          responseSchema: {
            type: "OBJECT",
            properties: {
              english: { type: "STRING" },
              chinese: { type: "STRING" }
            }
          }
        }
      };

      const result = await callGemini(payload);
      const dataText = result.candidates?.[0]?.content?.parts?.[0]?.text;
      if (!dataText) throw new Error("生成结果为空");
      
      const parsed = JSON.parse(dataText);
      setPromptEn(parsed.english || "");
      setPromptZh(parsed.chinese || "");
      
    } catch (err) {
      setError(err.message);
    } finally {
      setIsLoading(false);
    }
  };

  // 翻译与同步逻辑
  const syncTranslation = async (direction) => {
    setIsSyncing(direction);
    try {
      const sourceText = direction === 'en2zh' ? promptEn : promptZh;
      const instruction = direction === 'en2zh' 
        ? `将这段 AI 绘画英文提示词精准翻译为中文。保留技术参数（如 --v, 权重等）不翻译。只需返回纯文本。`
        : `将这段中文需求转化为地道的、适配 ${selectedModel.name} 的 AI 绘画英文提示词。保留技术参数不变。只需返回纯文本。`;
      
      const payload = {
        contents: [{ parts: [{ text: `${instruction}\n\n内容:\n${sourceText}` }] }]
      };
      
      const result = await callGemini(payload);
      const resText = result.candidates?.[0]?.content?.parts?.[0]?.text?.trim();
      
      if (resText) {
        if (direction === 'en2zh') setPromptZh(resText);
        else setPromptEn(resText);
      }
    } catch (err) {
      alert("同步翻译失败，请稍后重试");
    } finally {
      setIsSyncing(null);
    }
  };

  const copyToClipboard = (text) => {
    const textarea = document.createElement('textarea');
    textarea.value = text;
    document.body.appendChild(textarea);
    textarea.select();
    document.execCommand('copy');
    document.body.removeChild(textarea);
  };

  return (
    <div className="min-h-screen bg-[#f4f6f8] text-[#111827] font-sans p-4 md:p-6">
      <div className="max-w-[1400px] mx-auto">
        <header className="flex flex-col md:flex-row md:items-center justify-between mb-6 gap-4">
          <div className="flex items-center gap-3">
            <div className="p-2.5 bg-slate-900 rounded-xl shadow-lg">
              <ShieldCheck className="text-emerald-400" size={24} />
            </div>
            <div>
              <h1 className="text-xl font-black tracking-tighter uppercase">Prompt Architect <span className="text-emerald-600">Pro</span></h1>
              <p className="text-[10px] text-slate-500 font-bold uppercase tracking-[0.2em]">防出画技术 / 双语实时编辑</p>
            </div>
          </div>
        </header>

        <div className="grid grid-cols-1 lg:grid-cols-12 gap-6">
          {/* 左侧控制区 (缩减为 4 列，并使用抽屉) */}
          <div className="lg:col-span-4 space-y-4">
            
            {/* 抽屉 1：选择模型 */}
            <section className="bg-white rounded-[1.5rem] shadow-sm border border-slate-100 overflow-hidden">
              <button 
                onClick={() => setIsModelDrawerOpen(!isModelDrawerOpen)}
                className="w-full p-4 flex items-center justify-between bg-white hover:bg-slate-50 transition-colors"
              >
                <div className="flex items-center gap-2">
                  <Settings2 size={16} className="text-indigo-500" />
                  <span className="text-xs font-black uppercase text-slate-700">步骤 1: 引擎选择 ({selectedModel.name})</span>
                </div>
                {isModelDrawerOpen ? <ChevronUp size={16} className="text-slate-400"/> : <ChevronDown size={16} className="text-slate-400"/>}
              </button>
              
              {isModelDrawerOpen && (
                <div className="p-4 pt-0 grid grid-cols-2 gap-2 border-t border-slate-50">
                  {MODELS.map(model => (
                    <button
                      key={model.id}
                      onClick={() => { setSelectedModel(model); setIsModelDrawerOpen(false); }}
                      className={`p-2 rounded-xl border-2 transition-all text-center ${
                        selectedModel.id === model.id ? 'border-indigo-500 bg-indigo-50 font-black text-indigo-700' : 'border-slate-50 bg-slate-50 hover:border-slate-200 text-slate-600'
                      } text-[10px]`}
                    >
                      {model.name}
                    </button>
                  ))}
                </div>
              )}
            </section>

            {/* 抽屉 2：风格锁定 */}
            <section className="bg-emerald-950 rounded-[1.5rem] text-white shadow-lg border border-emerald-900 overflow-hidden">
              <button 
                onClick={() => setIsStyleDrawerOpen(!isStyleDrawerOpen)}
                className="w-full p-4 flex items-center justify-between hover:bg-emerald-900 transition-colors"
              >
                <div className="flex items-center gap-2">
                  <Palette size={16} className="text-emerald-400" />
                  <span className="text-xs font-black uppercase text-emerald-50">步骤 2: 风格锁定 ({styleMode.toUpperCase()})</span>
                </div>
                {isStyleDrawerOpen ? <ChevronUp size={16} className="text-emerald-400"/> : <ChevronDown size={16} className="text-emerald-400"/>}
              </button>
              
              {isStyleDrawerOpen && (
                <div className="p-4 pt-0 grid grid-cols-3 gap-2 border-t border-emerald-800/50">
                  {[
                    { id: '3d', label: '3D 写实', icon: <Box size={14}/> },
                    { id: '2d', label: '2D 动漫', icon: <ImageIcon size={14}/> },
                    { id: 'auto', label: '自动检测', icon: <Sparkles size={14}/> }
                  ].map(item => (
                    <button
                      key={item.id}
                      onClick={() => { setStyleMode(item.id); setIsStyleDrawerOpen(false); }}
                      className={`flex flex-col items-center gap-1 p-2 rounded-xl border transition-all ${
                        styleMode === item.id ? 'bg-emerald-500 text-slate-900 border-emerald-400' : 'bg-white/5 text-emerald-50 border-white/10 hover:bg-white/10'
                      }`}
                    >
                      {item.icon}
                      <span className="text-[10px] font-black">{item.label}</span>
                    </button>
                  ))}
                </div>
              )}
            </section>

            {/* 步骤 3：特征提取与附加提示词 (常驻显示) */}
            <section className="bg-white p-5 rounded-[1.5rem] shadow-sm border border-slate-100 flex flex-col gap-4">
              <div>
                <label className="text-[10px] font-black uppercase text-slate-400 mb-3 block px-1 flex items-center justify-between">
                  <span>步骤 3: 视觉特征提取</span>
                  <span className="text-[8px] bg-slate-100 px-2 py-0.5 rounded text-slate-500">已启用防出画保护</span>
                </label>
                <div className="grid grid-cols-2 gap-3">
                  {[
                    { key: 'clothing', label: '服装(屏蔽长相)', icon: '👕' },
                    { key: 'appearance', label: '面孔(屏蔽衣服)', icon: '👤' },
                    { key: 'layout', label: '模板(防裁切)', icon: '🖼️' },
                    { key: 'styleRef', label: '画风(屏蔽内容)', icon: '🎨' }
                  ].map((item) => (
                    <div key={item.key} className="relative group">
                      <input type="file" accept="image/*" className="hidden" id={`up-${item.key}`} onChange={(e) => handleImageUpload(e, item.key)} />
                      <label htmlFor={`up-${item.key}`} className={`aspect-[4/3] rounded-2xl border-2 border-dashed flex flex-col items-center justify-center cursor-pointer transition-all ${images[item.key] ? 'border-emerald-400 bg-emerald-50/30' : 'border-slate-200 hover:border-emerald-400 bg-slate-50'}`}>
                        {images[item.key] ? (
                          <img src={images[item.key]} className="w-full h-full object-cover rounded-[14px]" alt="Ref" />
                        ) : (
                          <>
                            <span className="text-xl mb-1">{item.icon}</span>
                            <span className="text-[9px] font-black text-slate-500 uppercase">{item.label}</span>
                          </>
                        )}
                      </label>
                      {images[item.key] && (
                        <button onClick={() => removeImage(item.key)} className="absolute -top-1.5 -right-1.5 bg-white text-red-500 p-1 rounded-full shadow-md border border-slate-100"><Trash2 size={10} /></button>
                      )}
                    </div>
                  ))}
                </div>
              </div>
              <div className="pt-2 border-t border-slate-100">
                <label className="text-[10px] font-black uppercase text-slate-400 mb-2 block px-1">
                  步骤 4: 附加要求文本 (可选)
                </label>
                <textarea
                  value={userText}
                  onChange={(e) => setUserText(e.target.value)}
                  placeholder="输入你想强加的元素，例如：角色拿着发光的剑、赛博朋克背景..."
                  className="w-full h-24 p-3 rounded-xl bg-slate-50 focus:bg-white focus:ring-2 focus:ring-emerald-100 outline-none transition-all text-xs font-medium resize-none"
                />
              </div>
            </section>

            <button
              onClick={generatePrompt}
              disabled={isLoading}
              className={`w-full py-4 rounded-[1.2rem] font-black text-white shadow-lg flex items-center justify-center gap-2 transition-all active:scale-[0.98] ${
                isLoading ? 'bg-slate-400 cursor-not-allowed' : 'bg-emerald-600 hover:bg-emerald-700 shadow-emerald-200'
              }`}
            >
              {isLoading ? <RefreshCw className="animate-spin" size={18} /> : <Zap size={18} />}
              {isLoading ? "正在生成双语提示词..." : `立即生成`}
            </button>
          </div>

          {/* 右侧输出区 (放大为 8 列，双语对照可编辑) */}
          <div className="lg:col-span-8 flex flex-col h-full">
            <div className="bg-white p-6 rounded-[2rem] shadow-sm border border-slate-100 flex-grow flex flex-col min-h-[500px]">
              <div className="text-xs font-black tracking-widest text-slate-400 uppercase mb-4 flex items-center justify-between px-1">
                <div className="flex items-center gap-2">
                  <Languages size={16} className="text-indigo-500" />
                  <span>Interactive Editor</span>
                </div>
                <div className="flex gap-1.5">
                  <div className="w-2 h-2 bg-indigo-500 rounded-full"></div>
                  <div className="w-2 h-2 bg-emerald-500 rounded-full"></div>
                </div>
              </div>
              
              {error ? (
                <div className="text-red-500 font-bold p-4 bg-red-50 rounded-2xl">{error}</div>
              ) : (
                <div className="flex flex-col lg:flex-row gap-4 flex-grow h-full">
                  
                  {/* English Col */}
                  <div className="flex-1 flex flex-col gap-2">
                    <label className="text-[10px] font-bold text-indigo-600 bg-indigo-50 px-3 py-1.5 rounded-lg w-max flex items-center gap-2">
                      <span className="w-1.5 h-1.5 rounded-full bg-indigo-600"></span> English Prompt
                    </label>
                    <textarea 
                      value={promptEn}
                      onChange={(e) => setPromptEn(e.target.value)}
                      placeholder="Waiting for generation..."
                      className="flex-grow w-full bg-[#f8fafc] border border-slate-100 rounded-2xl p-4 font-mono text-[11px] leading-relaxed text-slate-800 outline-none focus:ring-2 focus:ring-indigo-100 resize-none"
                    />
                    <div className="flex gap-2">
                      <button 
                        onClick={() => copyToClipboard(promptEn)}
                        disabled={!promptEn}
                        className="flex-1 py-2.5 bg-slate-900 text-white rounded-xl text-xs font-black hover:bg-indigo-600 transition-colors disabled:opacity-50 flex justify-center items-center gap-1.5"
                      >
                        <Copy size={12} /> 复制
                      </button>
                      <button 
                        onClick={() => syncTranslation('en2zh')}
                        disabled={!promptEn || isSyncing !== null}
                        className="flex-1 py-2.5 bg-indigo-50 text-indigo-600 border border-indigo-100 rounded-xl text-xs font-black hover:bg-indigo-100 transition-colors disabled:opacity-50 flex justify-center items-center gap-1.5"
                        title="将英文修改翻译到右侧中文"
                      >
                        {isSyncing === 'en2zh' ? <RefreshCw size={12} className="animate-spin" /> : <ArrowRightLeft size={12} />}
                        翻译为中文
                      </button>
                    </div>
                  </div>

                  {/* Divider line on desktop */}
                  <div className="hidden lg:block w-px bg-slate-100 my-8"></div>

                  {/* Chinese Col */}
                  <div className="flex-1 flex flex-col gap-2">
                    <label className="text-[10px] font-bold text-emerald-600 bg-emerald-50 px-3 py-1.5 rounded-lg w-max flex items-center gap-2">
                       <span className="w-1.5 h-1.5 rounded-full bg-emerald-600"></span> 中文对照解析
                    </label>
                    <textarea 
                      value={promptZh}
                      onChange={(e) => setPromptZh(e.target.value)}
                      placeholder="等待生成..."
                      className="flex-grow w-full bg-[#f8fafc] border border-slate-100 rounded-2xl p-4 font-sans text-xs leading-relaxed text-slate-700 outline-none focus:ring-2 focus:ring-emerald-100 resize-none"
                    />
                    <div className="flex gap-2">
                      <button 
                        onClick={() => copyToClipboard(promptZh)}
                        disabled={!promptZh}
                        className="flex-1 py-2.5 bg-white border border-slate-200 text-slate-700 rounded-xl text-xs font-black hover:bg-slate-50 transition-colors disabled:opacity-50 flex justify-center items-center gap-1.5"
                      >
                        <Copy size={12} /> 复制
                      </button>
                      <button 
                        onClick={() => syncTranslation('zh2en')}
                        disabled={!promptZh || isSyncing !== null}
                        className="flex-1 py-2.5 bg-emerald-50 text-emerald-600 border border-emerald-100 rounded-xl text-xs font-black hover:bg-emerald-100 transition-colors disabled:opacity-50 flex justify-center items-center gap-1.5"
                        title="将中文修改翻译回左侧英文"
                      >
                        {isSyncing === 'zh2en' ? <RefreshCw size={12} className="animate-spin" /> : <ArrowRightLeft size={12} />}
                        同步回英文
                      </button>
                    </div>
                  </div>

                </div>
              )}
            </div>
          </div>
        </div>
      </div>
    </div>
  );
};

export default App;
